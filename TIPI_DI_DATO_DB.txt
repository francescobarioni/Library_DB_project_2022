INFORMAZIONI SU TIPI DI DATO DEL DATABASE BIBLIOTECA:

1) ENTITA' AUTORE:
    - CODICE_AUTORE: INTERO,
    - LUOGO_NASCITA: VARCHAR 35,
    - NOME: VARCHAR 35,
    - COGNOME: VARCHAR 35,
    - DATA_NASCITA: DATE (yyyy-mm-dd);

2) ENTITA' UTENTE:
    - MATRICOLA: CHAR (6 CIFRE MAX),
    - NOME: VARCHAR 35,
    - COGNOME: VARCHAR 35,
    - INDIRIZZO: VARCHAR 60,
    - TELEFONO: INT 10 / CHAR 10;

3) ENTITA' LIBRO:
    - CODICE_LIBRO: INTERO,
    - TITOLO: VARCHAR 20,
    - ISBN: CHAR 13,
    - LINGUA: VARCHAR 20,
    - ANNO_PUBBLICAZIONE: INT 4
    - CODICE_BIBLIOTECA: INTERO,
    - NUMERO_COPIE: INTERO;

4) ENTITA' BIBLIOTECA:
    - CODICE_BIBLIOTECA: INTERO,
    - NOME: VARCHAR 35,
    - INDIRIZZO: VARCHAR 60;

5) ENTITA' DIPARTIMENTO:
    - CODICE_DIPARTIMENTO: INTERO,
    - PRESIDE: VARCHAR 35,
    - CODICE_BIBLIOTECA: INTERO;

6) RELAZIONE PRENDE_PRESTITO (N,M):
    - MATRICOLA: CHAR (6 CIFRE MAX),
    - CODICE_LIBRO: INTERO,
    - DATA_USCITA: DATE,
    - DATA_INGRESSO: DATE;

7) RELAZIONE SCRITTO_DA (N,M):
    - CODICE_LIBRO: INTERO,
    - CODICE_AUTORE: INTERO;
